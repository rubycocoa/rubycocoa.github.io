def applicationDidFinishLaunching(sender)
	begin
	  @proc.call
	rescue Exception => err
	  warn "#{err.message} (#{err.class})\n"
	  warn err.backtrace.join("\n    ")
	ensure
	  OSX::NSApplication.sharedApplication.terminate(self) if @terminate
	end
end
