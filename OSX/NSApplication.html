def NSApplication.run_with_temp_app(terminate = true, &proc)
  # prepare delegate
  delegate = RBCCTemporaryDelegate.alloc.init
  delegate.proc = proc
  delegate.terminate = terminate
  # run a new app
  app = NSApplication.sharedApplication
  app.setDelegate(delegate)
  app.run
end
