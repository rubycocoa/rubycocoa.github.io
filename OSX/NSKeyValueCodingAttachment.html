def rbSetValue_forKey(value, key)
  if m = kvc_setter_method(key.to_s)
    send(m, value)
  else
    kvc_accessor_notfound(key)
  end
end

def rbValueForKey(key)
  if m = kvc_getter_method(key.to_s)
    return send(m)
  else
    kvc_accessor_notfound(key)
  end
end
